// Login — מחזיר שני Tokens app.post('/auth/login', async (req, res) => { const user = await User.findByEmail(req.body.email); if (!user || !await bcrypt.compare(req.body.password, user.hash)) return res.status(401).json({ title: 'Invalid credentials' }); // Access Token — קצר, ב-JSON body const accessToken = jwt.sign( { sub: user.id, roles: user.roles }, process.env.ACCESS_SECRET, { expiresIn: '15m', algorithm: 'RS256' } ); // Refresh Token — ארוך, ב-HttpOnly Cookie בלבד const refreshToken = jwt.sign( { sub: user.id, tokenVersion: user.tokenVersion }, process.env.REFRESH_SECRET, { expiresIn: '7d' } ); res.cookie('refreshToken', refreshToken, { httpOnly: true, ← JavaScript לא יכול לקרוא secure: true, ← HTTPS בלבד sameSite: 'strict', ← מגן מ-CSRF maxAge: 7 * 24 * 60 * 60 * 1000 }).json({ accessToken }); });
05
JWT שנשתנה: Signature לא תואמת → JsonWebTokenError: invalid signature → 401. JWT לא מוצפן אבל חתום — שינוי Payload שובר את ה-Signature. RS256 > HS256 כי Private Key לא חשוף ל-Clients.
04
Token Rotation: כל Refresh שולח Refresh Token חדש ומבטל את הישן ב-Redis. אם מישהו גנב Refresh Token ישן — הוא כבר Blacklisted. One-Time Use בפועל.
⚠️ לעולם אל תשמור Access Token ב-localStorage — XSS יגנוב אותו בשנייה. שמור Access Token ב-Memory (JavaScript Variable). Refresh Token רק ב-HttpOnly Cookie. זה ה-Pattern של Auth0, Supabase, NextAuth.

// Server Events io.use(async (socket, next) => { try { const decoded = jwt.verify(socket.handshake.auth.token, SECRET); socket.userId = decoded.sub; next(); } catch { next(new Error('Unauthorized')); } }); io.on('connection', (socket) => { socket.on('join-room', (roomId) => { socket.join(roomId); socket.to(roomId).emit('user-joined', { userId: socket.userId }); }); socket.on('send-message', ({ roomId, text }) => { const msg = { text, from: socket.userId, at: Date.now() }; io.to(roomId).emit('message', msg); ← io.to = כולל שולח, socket.to = רק אחרים }); // Typing with auto-stop let typingTimer; socket.on('typing-start', (roomId) => { socket.to(roomId).emit('user-typing', socket.userId); clearTimeout(typingTimer); typingTimer = setTimeout(() => socket.to(roomId).emit('user-stopped-typing', socket.userId), 2000); }); });
05
למה Redis Adapter? Server 1 מחזיק Socket של User A. Server 2 מחזיק Socket של User B. בלי Redis — message מServer 1 לא מגיע ל-User B. עם Redis Pub/Sub → כל Server שומע Events → מעביר ל-Sockets שהוא מחזיק.
💡 socket.to(room) = כל המחוברים לRoom חוץ מהשולח. io.to(room) = כולל השולח. socket.broadcast = כולם חוץ מהשולח בכל Rooms. הבדל קטן, bug גדול אם מתבלבלים.

✅ Rate Limiting + ETag Implementation
// Rate Limiter import rateLimit from 'express-rate-limit'; import RedisStore from 'rate-limit-redis'; const limiter = rateLimit({ windowMs: 15 * 60 * 1000, max: 100, standardHeaders: 'draft-7', ← RateLimit headers RFC legacyHeaders: false, store: new RedisStore({ client: redisClient }), handler: (req, res) => { res.status(429).json({ type: 'https://api.co/errors/rate-limit', title: 'Too Many Requests', status: 429, retryAfter: Math.ceil((req.rateLimit.resetTime - Date.now()) / 1000) }); } }); // ETag Middleware const etagMiddleware = (req, res, next) => { const originalJson = res.json.bind(res); res.json = (data) => { const etag = `"${crypto.createHash('md5').update(JSON.stringify(data)).digest('hex')}"`; res.setHeader('ETag', etag); if (req.headers['if-none-match'] === etag) return res.status(304).end(); ← חסכנו כל ה-Bandwidth! return originalJson(data); }; next(); };
05
שאילתות 1-100 → 200 OK עם headers X-RateLimit-Remaining: 99...0. שאילתה 101 → 429 Too Many Requests. Redis מחזיק Counter לכל IP עם TTL של 15 דקות.
💡 ETag vs max-age: max-age = "אל תשאל ל-N שניות". ETag = "שאל, אבל אם לא השתנה קבל 304 ללא Body". ETag תמיד fresh. max-age יכול להגיש stale. stale-while-revalidate = ה-Best of Both Worlds.

// k6-script.js import http from 'k6/http'; import { check, sleep } from 'k6'; export const options = { stages: [ { duration: '2m', target: 50 }, // Ramp-up to 50 VUs { duration: '3m', target: 50 }, // Steady at 50 { duration: '2m', target: 200 }, // Spike to 200 { duration: '3m', target: 200 }, // Steady at 200 { duration: '2m', target: 0 }, // Ramp-down ], thresholds: { 'http_req_duration': ['p(95)<500'], // p95 < 500ms 'http_req_failed': ['rate<0.01'], // < 1% errors }, }; export default function () { const res = http.get('http://localhost:3000/api/users'); check(res, { 'status 200': (r) => r.status === 200, 'body not empty': (r) => r.body.length > 0, }); sleep(1); // VU מחכה שנייה בין בקשות (think time) }
💡
Bottleneck Analysis:
p95 עולה בSteady State → CPU Bound → הוסף CPU/Workers
p95 מתפוצץ בSpike → DB Connections → הגדל Connection Pool
Memory גדל ולא יורד → Memory Leak → Heap Profiling
Errors פתאומיים → Rate Limit שפגעת בו → הגדל Limits
💡 p95 vs Average: Average מסתיר בעיות. p95 = 95% מהRequests היו מהירים מזה. 5% האיטיים = ה-Users הכי ממורמרים. בפרודקשן תמיד עבוד עם Percentiles, לא Averages.
⚠️ הרץ Load Tests על Environment שאינו Production. אם חייב Production — הרץ בשעות לא שיא, עם gradual ramp-up, ויד על Stop Button.
